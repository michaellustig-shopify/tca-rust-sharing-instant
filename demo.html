Rust — Nested Query via Fetch

    
// FetchKeyRequest combines multiple queries into one atomic observation

struct RecordingDetailRequest {
    memo_id: String,
}

impl FetchKeyRequest for RecordingDetailRequest {
    type Value = RecordingDetail;

    fn fetch(&self, db: &dyn Database) -> Result<RecordingDetail> {
        // Single nested query — InstantDB resolves links server-side
        let q = json_to_value(&serde_json::json!({
            "recordings": {
                "$": { "where": { "memo_id": self.memo_id } },
                "segments": {
                    "ideas": {},       // nested: segment → ideas
                    "tasks": {},       // nested: segment → tasks
                    "persons": {},     // nested: segment → persons
                }
            }
        }));
        let result = db.query(&q)?;
        // Parse nested response into typed structs...
    }
}

// Usage — just like @Fetch in Swift:
let detail = Fetch::new(
    RecordingDetailRequest { memo_id: "memo-42".into() },
    db.clone(),
);

// Reactive! Updates when ANY linked entity changes
let recording = detail.get();
for seg in &recording.segments {
    for idea in &seg.ideas {
        println!("  Idea: {} ({})", idea.name, idea.status);
    }
}


    
Rust — Shared State with Links

    
// Idea evolves into Product — relationship tracked reactively

let idea = Shared::new(
    Idea {
        slug: "voice-activated-dog-treat-cannon".into(),
        name: "Voice-Activated Dog Treat Cannon".into(),
        status: "conceptual".into(),
        ..
    },
    InstantDbKey::new("idea-treat-cannon", db.clone()),
);

// Mutate — auto-persists to InstantDB, syncs to all clients
idea.with_lock(|i| {
    i.status = "prototyped".into();
});

// Watch for changes from other clients via WebSocket
let mut rx = idea.watch();
while rx.changed().await.is_ok() {
    println!("Idea updated: {:?}", *rx.borrow());
}
